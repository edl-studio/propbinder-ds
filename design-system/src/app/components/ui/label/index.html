Angular Component Usage

<!-- Basic Label -->
<ds-label for="email-input">Email Address</ds-label>
<ds-input id="email-input" placeholder="Enter email"></ds-input>

<!-- Label with Custom Classes -->
<ds-label className="font-semibold text-success" for="success-input">
  Valid Input
</ds-label>

<!-- Label with ID -->
<ds-label id="form-label" for="form-input" className="required">
  Required Field
</ds-label>

HTML Structure

<label class="ui-sm-regular block" for="input-id">
  Label Text
</label>
<input id="input-id" type="text" placeholder="Input field">
